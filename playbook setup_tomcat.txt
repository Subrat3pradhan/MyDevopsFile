---
- name: This playbook for setup tomcate
  hosts: all
  become: yes

  tasks:
  - name: install java on RedHat
    yum:
      name: java
      state: present
    when: ansible_os_family == "RedHat"

  - name: install java on Debian
    apt:
      name: default-jdk
      state: present
    when: ansible_os_family == "Debian"

  - name: download tomcate
    get_url:
      url: https://dlcdn.apache.org/tomcat/tomcat-9/v9.0.98/bin/apache-tomcat-9.0.98.tar.gz
      dest: /opt

  - name: Extracting tomcate
    unarchive:
      src: /opt/apache-tomcat-9.0.98.tar.gz
      dest: /opt
      remote_src: yes

  - name: change file permission
    file:
      path: /opt/apache-tomcat-9.0.98/bin/startup.sh
      mode: 0777

  - name: start tomcart service
    shell: nohup ./startup.sh
    args:
      chdir: /opt/apache-tomcat-9.0.98/bin
